> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a premium, bank-grade Next.js 15 frontend with two interfaces — a chat-first customer portal and an officer HITL dashboard — connected to the FastAPI backend via SSE streaming.

**Tech Stack:** Next.js 15, React 19, TypeScript, Tailwind CSS, shadcn/ui, Vercel AI SDK (`@ai-sdk/react`), Auth.js v5, Framer Motion, Lucide React

**Spec:** `docs/superpowers/specs/2026-04-16-v4-langgraph-sso-frontend-design.md`

**Backend:** Already built — FastAPI at `http://localhost:8000` with SSE at `POST /api/v1/chat/stream`, auth at `/api/v1/auth/*`, HITL at `/api/v1/hitl/*`

```typescript
import NextAuth from "next-auth"
import Google from "next-auth/providers/google"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Google],
  callbacks: {
    async jwt({ token, account }) {
      // On first sign-in, call our backend to create/find user and get our JWT
      if (account) {
        const res = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/api/v1/auth/google/callback?code=${account.access_token}`, {
          method: "GET",
        }).catch(() => null);

        if (res?.ok) {
          // Backend redirects, but we extract the token from the redirect URL
          const redirectUrl = res.url;
          const url = new URL(redirectUrl);
          const backendToken = url.searchParams.get("token");
          if (backendToken) {
            token.backendToken = backendToken;
          }
        }
      }
      return token;
    },
    async session({ session, token }) {
      // Expose backend token to client
      session.backendToken = token.backendToken as string;
      return session;
    },
  },
  pages: {
    signIn: "/",
  },
})
```

```typescript
export default auth((req) => {
  const isLoggedIn = !!req.auth;
  const isProtectedRoute = req.nextUrl.pathname.startsWith("/chat") ||
                            req.nextUrl.pathname.startsWith("/dashboard") ||
                            req.nextUrl.pathname.startsWith("/status");

    return Response.redirect(new URL("/", req.nextUrl.origin));
  }
})
```

```typescript
const API_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

export async function apiClient(path: string, options: RequestInit & { token?: string } = {}) {
  const { token, ...fetchOptions } = options;
  const headers = new Headers(fetchOptions.headers);
  if (token) {
    headers.set("Authorization", `Bearer ${token}`);
  }
  headers.set("Content-Type", "application/json");

  const res = await fetch(`${API_URL}${path}`, {
    ...fetchOptions,
    headers,
  });

  if (!res.ok) {
    throw new Error(`API error: ${res.status} ${res.statusText}`);
  }

  return res.json();
}

export { API_URL };
```

```tsx
"use client";

import { useState, useRef, KeyboardEvent } from "react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { SendHorizontal } from "lucide-react";

interface ChatInputProps {
  onSend: (message: string) => void;
  disabled?: boolean;
  placeholder?: string;
}

export function ChatInput({ onSend, disabled, placeholder = "Type a message..." }: ChatInputProps) {
  const [value, setValue] = useState("");
  const textareaRef = useRef<HTMLTextAreaElement>(null);

  const handleSend = () => {
    const trimmed = value.trim();
    if (!trimmed || disabled) return;
    onSend(trimmed);
    setValue("");
  };

  const handleKeyDown = (e: KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  return (
    <div className="border-t border-slate-200 bg-white p-4">
      <div className="max-w-3xl mx-auto flex items-end gap-3">
        <Textarea
          ref={textareaRef}
          value={value}
          onChange={(e) => setValue(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder={placeholder}
          disabled={disabled}
          className="min-h-[44px] max-h-[120px] resize-none rounded-xl border-slate-300 focus:border-[#0F172A] focus:ring-[#0F172A]"
          rows={1}
        />
        <Button
          onClick={handleSend}
          disabled={disabled || !value.trim()}
          size="icon"
          className="h-11 w-11 rounded-xl bg-[#0F172A] hover:bg-[#1E293B] shrink-0"
        >
          <SendHorizontal className="w-4 h-4" />
        </Button>
      </div>
    </div>
  );
}
```

```tsx
"use client";

import { useChat } from "@ai-sdk/react";
import { DefaultChatTransport } from "ai";
import { useSession } from "next-auth/react";
import { useRef, useEffect } from "react";
import { ScrollArea } from "@/components/ui/scroll-area";
import { MessageBubble } from "./message-bubble";
import { ChatInput } from "./chat-input";
import { ToolRenderer } from "./tool-renderer";

const API_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

export function ChatInterface() {
  const { data: session } = useSession();
  const scrollRef = useRef<HTMLDivElement>(null);

  const { messages, sendMessage, isLoading } = useChat({
    transport: new DefaultChatTransport({
      api: `${API_URL}/api/v1/chat/stream`,
      headers: {
        Authorization: `Bearer ${(session as any)?.backendToken || ""}`,
      },
    }),
  });

  // Auto-scroll to bottom on new messages
  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages]);

  const handleSend = (text: string) => {
    sendMessage({ text });
  };

  return (
    <div className="flex-1 flex flex-col overflow-hidden">
      {/* Messages area */}
      <ScrollArea ref={scrollRef} className="flex-1 px-6 py-4">
        <div className="max-w-3xl mx-auto space-y-6">
          {messages.length === 0 && (
            <div className="text-center py-20">
              <div className="w-16 h-16 rounded-2xl bg-[#D4A853]/10 flex items-center justify-center mx-auto mb-4">
                <span className="text-[#D4A853] text-2xl">AI</span>
              </div>
              <h2 className="text-xl font-semibold text-slate-800 mb-2">
                Welcome to LoanAI
              </h2>
              <p className="text-slate-500 max-w-md mx-auto">
                I'll guide you through your personal loan application step by step.
                Let's get started!
              </p>
            </div>
          )}

          {messages.map((message) => (
            <MessageBubble key={message.id} role={message.role as "user" | "assistant"}>
              {message.parts.map((part, i) => {
                if (part.type === "text") {
                  return <p key={`${message.id}-${i}`} className="whitespace-pre-wrap text-sm leading-relaxed">{part.text}</p>;
                }
                // Render tool/data parts via ToolRenderer
                if (part.type.startsWith("data-tool-")) {
                  const toolName = part.type.replace("data-tool-", "");
                  return <ToolRenderer key={`${message.id}-${i}`} toolName={toolName} data={(part as any).data} />;
                }
                if (part.type.startsWith("data-status")) {
                  return (
                    <p key={`${message.id}-${i}`} className="text-xs text-slate-400 italic">
                      Processing: {(part as any).data?.phase}
                    </p>
                  );
                }
                return null;
              })}
            </MessageBubble>
          ))}

          {isLoading && (
            <MessageBubble role="assistant">
              <div className="flex items-center gap-2">
                <div className="flex gap-1">
                  <span className="w-2 h-2 bg-[#D4A853] rounded-full animate-bounce [animation-delay:0ms]" />
                  <span className="w-2 h-2 bg-[#D4A853] rounded-full animate-bounce [animation-delay:150ms]" />
                  <span className="w-2 h-2 bg-[#D4A853] rounded-full animate-bounce [animation-delay:300ms]" />
                </div>
                <span className="text-sm text-slate-400">Processing...</span>
              </div>
            </MessageBubble>
          )}
        </div>
      </ScrollArea>

      {/* Input */}
      <ChatInput onSend={handleSend} disabled={isLoading} />
    </div>
  );
}
```

```tsx
"use client";

import { useState, useRef } from "react";
import { motion } from "framer-motion";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Upload, FileCheck, Loader2 } from "lucide-react";

interface DocUploadWidgetProps {
  documentType: string;
  label: string;
  onUpload: (file: File) => void;
}

export function DocUploadWidget({ documentType, label, onUpload }: DocUploadWidgetProps) {
  const [file, setFile] = useState<File | null>(null);
  const [uploading, setUploading] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);

  const handleFile = (f: File) => {
    setFile(f);
    setUploading(true);
    onUpload(f);
    setTimeout(() => setUploading(false), 1500);
  };

  return (
    <motion.div initial={{ opacity: 0, y: 4 }} animate={{ opacity: 1, y: 0 }}>
      <Card className="border-dashed border-2 border-[#D4A853]/40 hover:border-[#D4A853] transition-colors cursor-pointer"
        onClick={() => inputRef.current?.click()}>
        <CardContent className="flex items-center gap-4 py-4">
          <div className="w-12 h-12 rounded-xl bg-[#D4A853]/10 flex items-center justify-center shrink-0">
            {uploading ? (
              <Loader2 className="w-5 h-5 text-[#D4A853] animate-spin" />
            ) : file ? (
              <FileCheck className="w-5 h-5 text-green-600" />
            ) : (
              <Upload className="w-5 h-5 text-[#D4A853]" />
            )}
          </div>
          <div className="flex-1 min-w-0">
            <p className="font-medium text-sm text-[#0F172A]">{label}</p>
            <p className="text-xs text-slate-400 mt-0.5">
              {file ? file.name : "Click to upload or drag and drop"}
            </p>
          </div>
          {!file && (
            <Button variant="outline" size="sm" className="shrink-0 text-xs">
              Browse
            </Button>
          )}
          <input ref={inputRef} type="file" className="hidden" accept="image/*,.pdf"
            onChange={(e) => e.target.files?.[0] && handleFile(e.target.files[0])} />
        </CardContent>
      </Card>
    </motion.div>
  );
}
```

```tsx
"use client";

import { useEffect, useState } from "react";
import { useSession } from "next-auth/react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import Link from "next/link";
import { Clock, AlertTriangle } from "lucide-react";
import { apiClient } from "@/lib/api";

interface QueueItem {
  application_id: string;
  escalation_reason: string;
  priority: string;
  agent_recommendation: string;
}

export function ReviewQueue() {
  const { data: session } = useSession();
  const [items, setItems] = useState<QueueItem[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!session) return;
    apiClient("/api/v1/hitl/queue", { token: (session as any).backendToken })
      .then((data) => setItems(data.items || []))
      .catch(() => setItems([]))
      .finally(() => setLoading(false));
  }, [session]);

  if (loading) return <div className="p-8 text-center text-slate-400">Loading queue...</div>;

  return (
    <div className="space-y-3">
      {items.length === 0 ? (
        <Card>
          <CardContent className="py-12 text-center">
            <Clock className="w-8 h-8 text-slate-300 mx-auto mb-3" />
            <p className="text-slate-500">No applications pending review</p>
          </CardContent>
        </Card>
      ) : (
        items.map((item) => (
          <Link key={item.application_id} href={`/dashboard/${item.application_id}`}>
            <Card className="hover:border-[#D4A853]/50 hover:shadow-md transition-all cursor-pointer">
              <CardContent className="flex items-center gap-4 py-4">
                <AlertTriangle className="w-5 h-5 text-amber-500 shrink-0" />
                <div className="flex-1 min-w-0">
                  <p className="font-medium text-sm truncate">{item.application_id.slice(0, 8)}...</p>
                  <p className="text-xs text-slate-500">{item.escalation_reason}</p>
                </div>
                <Badge variant="outline" className="shrink-0">{item.priority}</Badge>
                <Badge className="shrink-0 bg-amber-100 text-amber-700">{item.agent_recommendation}</Badge>
              </CardContent>
            </Card>
          </Link>
        ))
      )}
    </div>
  );
}
```

```tsx
    apiClient(`/api/v1/hitl/${id}`, { token: (session as any).backendToken })
      .then(setDetail)
      .catch(() => setDetail(null))
      .finally(() => setLoading(false));
  }, [session, id]);

  const handleAction = async (decision: string, notes: string) => {
    await apiClient(`/api/v1/hitl/${id}/review`, {
      method: "POST",
      token: (session as any).backendToken,
      body: JSON.stringify({ decision, officer_notes: notes }),
    });
    window.location.href = "/dashboard";
  };

  if (loading) return <div className="p-8 text-center text-slate-400">Loading...</div>;
  if (!detail) return <div className="p-8 text-center text-red-500">Application not found</div>;

  return (
    <div className="flex-1 flex flex-col overflow-hidden">
      <div className="flex-1 overflow-auto p-6">
        <div className="max-w-5xl mx-auto grid grid-cols-2 gap-6">
          {/* Application Info */}
          <Card>
            <CardHeader>
              <CardTitle className="text-base">Application</CardTitle>
            </CardHeader>
            <CardContent className="space-y-2 text-sm">
              <div><span className="text-slate-400">Name:</span> {detail.full_application_data.applicant_name}</div>
              <div><span className="text-slate-400">Amount:</span> {detail.full_application_data.loan_amount} INR</div>
              <div><span className="text-slate-400">Type:</span> {detail.full_application_data.loan_type}</div>
              <div><span className="text-slate-400">Status:</span> <Badge variant="outline">{detail.full_application_data.status}</Badge></div>
            </CardContent>
          </Card>

          {/* AI Recommendation */}
          <Card>
            <CardHeader>
              <CardTitle className="text-base">AI Recommendation</CardTitle>
            </CardHeader>
            <CardContent>
              <Badge className="bg-amber-100 text-amber-700 text-sm">{detail.recommended_decision}</Badge>
              {detail.risk_flags.length > 0 && (
                <div className="mt-3 space-y-1">
                  <p className="text-xs text-slate-400">Risk Flags:</p>
                  {detail.risk_flags.map((flag: string, i: number) => (
                    <p key={i} className="text-sm text-red-600">- {flag}</p>
                  ))}
                </div>
              )}
            </CardContent>
          </Card>

          {/* Agent Outputs */}
          <Card className="col-span-2">
            <CardHeader>
              <CardTitle className="text-base">Agent Outputs</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="space-y-3">
                {detail.agent_outputs.map((output: any, i: number) => (
                  <div key={i} className="p-3 bg-slate-50 rounded-lg">
                    <p className="text-xs text-slate-400 font-medium uppercase">{output.agent}</p>
                    <pre className="text-xs text-slate-600 mt-1 overflow-auto">{JSON.stringify(output.output, null, 2)}</pre>
                  </div>
                ))}
              </div>
            </CardContent>
          </Card>
        </div>
      </div>

      <ActionBar onAction={handleAction} />
    </div>
  );
}
```
